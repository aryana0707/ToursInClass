CREATE TABLE  "USERREGISTER" 
   (	"FIRST_NAME" VARCHAR2(200), 
	"LAST_NAME" VARCHAR2(200), 
	"EMAIL" VARCHAR2(200) NOT NULL ENABLE, 
	"MOBILE" VARCHAR2(200), 
	"PASSWORD" VARCHAR2(200), 
	"GENDER" VARCHAR2(50), 
	"ADDRESS" VARCHAR2(250), 
	"DOB" DATE, 
	"DOREGISTRATION" VARCHAR2(200), 
	 PRIMARY KEY ("EMAIL") ENABLE
   )
/
